program SomaDoisNumeros;

var
  numero1, numero2, resultado: integer;

begin
  write('Digite o primeiro número: ');
  readln(numero1);

  write('Digite o segundo número: ');
  readln(numero2);

  resultado := numero1 + numero2;

  writeln;
  writeln('A soma de ', numero1, ' + ', numero2, ' é igual a: ', resultado);

  readln;
end.
